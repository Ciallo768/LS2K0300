void DetectRedBlock(cv::Mat &src)
{
    red_objects.clear();
    red_area = 0;
    if (src.empty()) return;

    // 只搜索下半部分，可根据实际改成 src.rows * 2 / 3
    const int roi_y = src.rows / 2;
    cv::Rect search_roi(0, roi_y, src.cols, src.rows - roi_y);
    cv::Mat src_roi = src(search_roi);

    static cv::Mat mask;
    static std::vector<std::vector<cv::Point>> contours;
    contours.clear();

    // 分配二值图
    mask.create(src_roi.rows, src_roi.cols, CV_8UC1);

    // 直接在 BGR 上判红
    for (int y = 0; y < src_roi.rows; ++y)
    {
        const cv::Vec3b* src_ptr = src_roi.ptr<cv::Vec3b>(y);
        uchar* mask_ptr = mask.ptr<uchar>(y);

        for (int x = 0; x < src_roi.cols; ++x)
        {
            int b = src_ptr[x][0];
            int g = src_ptr[x][1];
            int r = src_ptr[x][2];

            if (r > 120 && (r - g) > 35 && (r - b) > 35)
                mask_ptr[x] = 255;
            else
                mask_ptr[x] = 0;
        }
    }

    // 如果噪声比较多，可以尝试开一下，但先别急着加
    // static cv::Mat kernel = cv::getStructuringElement(cv::MORPH_RECT, cv::Size(3, 3));
    // cv::morphologyEx(mask, mask, cv::MORPH_OPEN, kernel);

    cv::findContours(mask, contours, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);

    int bottom_y = -1;
    cv::Point bestCenter;
    cv::Rect bestRect;

    for (auto &c : contours)
    {
        cv::Rect rect = cv::boundingRect(c);
        int area = rect.area();

        if (area < MIN_RED_AREA) continue;

        int cx = rect.x + rect.width / 2;
        int cy = rect.y + rect.height / 2;

        if (cy > bottom_y)
        {
            bottom_y = cy;
            bestCenter = cv::Point(cx, cy);
            bestRect = rect;
        }
    }

    if (bottom_y >= 0)
    {
        bestCenter.y += roi_y;
        bestRect.y += roi_y;

        red_area = bestRect.area();
        red_objects.push_back({bestCenter, red_area});
    }
}

    static std::vector<std::vector<cv::Point>> contours;
    contours.clear();

    // 分配二值图
    mask.create(src_roi.rows, src_roi.cols, CV_8UC1);

    // 直接在 BGR 上判红
    for (int y = 0; y < src_roi.rows; ++y)
    {
        const cv::Vec3b* src_ptr = src_roi.ptr<cv::Vec3b>(y);
        uchar* mask_ptr = mask.ptr<uchar>(y);

        for (int x = 0; x < src_roi.cols; ++x)
        {
            int b = src_ptr[x][0];
            int g = src_ptr[x][1];
            int r = src_ptr[x][2];
            int sum_gb = g+b;
            if (r > 90 && (r > 25+g) && r > (b+25)&&r*2>g+b+80)
                mask_ptr[x] = 255;
            else
                mask_ptr[x] = 0;
        }
    }

    // 如果噪声比较多，可以尝试开一下，但先别急着加
    // static cv::Mat kernel = cv::getStructuringElement(cv::MORPH_RECT, cv::Size(3, 3));
    // cv::morphologyEx(mask, mask, cv::MORPH_OPEN, kernel);

    cv::findContours(mask, contours, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);

    int bottom_y = -1;
    cv::Point bestCenter;
    cv::Rect bestRect;

    for (auto &c : contours)
    {
        cv::Rect rect = cv::boundingRect(c);
        int area = rect.area();

        if (area < MIN_RED_AREA) continue;

        int cx = rect.x + rect.width / 2;
        int cy = rect.y + rect.height / 2;

        if (cy > bottom_y)
        {
            bottom_y = cy;
            bestCenter = cv::Point(cx, cy);
            bestRect = rect;
        }
    }

    if (bottom_y >= 0)
    {
        bestCenter.y += roi_y;
        bestRect.y += roi_y;

        red_area = bestRect.area();
        red_objects.push_back({bestCenter, red_area});
        //printf("cx:%d,cy:%d,w:%d,h:%d,area:%d\n",bestCenter.x,bestCenter.y,bestRect.width,bestRect.height,red_area);
    }
}

void ICM_AHRSupdate(float gx, float gy, float gz, float ax, float ay, float az) {
    float halfT = 0.5 * delta_T;
    float vx, vy, vz;
    float ex, ey, ez;
    float q0 = Q_info.q0;
    float q1 = Q_info.q1;
    float q2 = Q_info.q2;
    float q3 = Q_info.q3;
    float q0q0 = q0 * q0;
    float q0q1 = q0 * q1;
    float q0q2 = q0 * q2;
//    float q0q3 = q0 * q3;
    float q1q1 = q1 * q1;
//    float q1q2 = q1 * q2;
    float q1q3 = q1 * q3;
    float q2q2 = q2 * q2;
    float q2q3 = q2 * q3;
    float q3q3 = q3 * q3;
    // float delta_2 = 0;

    float norm = fast_sqrt(ax * ax + ay * ay + az * az);
    ax = ax * norm;
    ay = ay * norm;
    az = az * norm;

    vx = 2 * (q1q3 - q0q2);
    vy = 2 * (q0q1 + q2q3);
    vz = q0q0 - q1q1 - q2q2 + q3q3;
    //vz = (q0*q0-0.5f+q3 * q3) * 2;


    ex = ay * vz - az * vy;
    ey = az * vx - ax * vz;
    ez = ax * vy - ay * vx;


    I_ex += halfT * ex;   // integral error scaled by Ki
    I_ey += halfT * ey;
    I_ez += halfT * ez;

    gx = gx + param_Kp * ex + param_Ki * I_ex;
    gy = gy + param_Kp * ey + param_Ki * I_ey;
    gz = gz + param_Kp * ez + param_Ki * I_ez;



    q0 = q0 + (-q1 * gx - q2 * gy - q3 * gz) * halfT;
    q1 = q1 + (q0 * gx + q2 * gz - q3 * gy) * halfT;
    q2 = q2 + (q0 * gy - q1 * gz + q3 * gx) * halfT;
    q3 = q3 + (q0 * gz + q1 * gy - q2 * gx) * halfT;



    // normalise quaternion
    norm = fast_sqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
    Q_info.q0 = q0 * norm;
    Q_info.q1 = q1 * norm;
    Q_info.q2 = q2 * norm;
    Q_info.q3 = q3 * norm;
}



void ICM_getEulerianAngles(void)
{
    imu660ra_get_gyro();
    imu660ra_get_acc();                                     // 获取 IMU660RA 加速度计数据


    ICM_getValues();
    ICM_AHRSupdate(icm_data.gyro_x, icm_data.gyro_y, icm_data.gyro_z, icm_data.acc_x, icm_data.acc_y, icm_data.acc_z);
    float q0 = Q_info.q0;
    float q1 = Q_info.q1;
    float q2 = Q_info.q2;
    float q3 = Q_info.q3;


    icm_data.pitch = asin(-2 * q1 * q3 + 2 * q0 * q2) * 180 / M_PI; // pitch
    icm_data.roll = atan2(2 * q2 * q3 + 2 * q0 * q1, -2 * q1 * q1 - 2 * q2 * q2 + 1) * 180 / M_PI; // roll
    icm_data.yaw = atan2(2 * q1 * q2 + 2 * q0 * q3, -2 * q2 * q2 - 2 * q3 * q3 + 1) * 180 / M_PI;//转向角
}

{
    red_objects.clear();//先清元素
    red_area = 0;
    if(src.empty()) return;
    //改为只搜索下半部分 
    const int roi_y = 0;
    cv::Rect search_roi(0,roi_y,src.cols,src.rows-roi_y);
    cv::Mat src_roi = src(search_roi);

    static cv::Mat hsv,mask1,mask2,mask;
    cv::Rect roi;

    static std::vector<std::vector<cv::Point>> contours;//存储每个色块的质心坐标的动态数组
    contours.clear();

    cv::cvtColor(src_roi,hsv,cv::COLOR_BGR2HSV);

    cv::inRange(hsv, cv::Scalar(0, 120, 70), cv::Scalar(10, 255, 255), mask1);//红色阈值
    cv::inRange(hsv, cv::Scalar(160, 120, 70), cv::Scalar(179, 255, 255), mask2);//红色阈值

    mask = mask1 | mask2;    

    cv::findContours(mask, contours, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);//找色块 (外轮廓)

    //int maxArea = 0;
    int bottom_y = 0;
    cv::Point bestCenter;//存储轮廓的质心坐标
    cv::Rect bestRect;//存储最小外接矩形


    //改为寻找最靠近的红色矩形块
    for(auto &c : contours){
        cv::Rect rect = cv::boundingRect(c);

        int area = rect.area();
        if(area < MIN_RED_AREA) continue;//排除红色噪点 （后续可以用于排除一些红色色块在远端的情况)
        cv::Moments m = cv::moments(c);
        if(m.m00 == 0) continue;
        
        int cx = rect.x + rect.width/2;
        int cy = rect.y + rect.height/2;

        if(cy > bottom_y){

            bottom_y =  cy;

            bestCenter = cv::Point(cx,cy);

            bestRect = rect;
        }
    }

    if(bottom_y > 0){
        bestCenter.y += roi_y;
        bestRect.y += roi_y;
        red_area = bestRect.area();

        red_objects.push_back({bestCenter,red_area});//存储最近红色矩形块的坐标和面积

    //    cv::rectangle(src, bestRect, cv::Scalar(0, 255, 0), 1);//绘制最小外接矩形

    //    printf("cx:%d,cy:%d,w:%d,h:%d,area:%d\n",bestCenter.x,bestCenter.y,bestRect.width,bestRect.height,red_area);
    }
